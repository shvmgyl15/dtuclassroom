#Oracle 10G

CREATE TABLE USERS(UNAME VARCHAR2(64) PRIMARY KEY NOT NULL, PWD VARCHAR2(170), NAME VARCHAR(64), SEX CHAR, DOB DATE, EMAIL VARCHAR2(320) UNIQUE, MOBILE_NO NUMBER(10,0), D CHAR)

CREATE TABLE STUDENTS(UNAME VARCHAR2(64), YEAR NUMBER(4,0), BATCH VARCHAR2(3), BNO NUMBER(3,0), ROLLNO NUMBER(3), constraint PK_D primary key (YEAR, BATCH, BNO, ROLLNO))

CREATE TABLE TEACHERS(TNAME VARCHAR2(64), BNAME VARCHAR2(64), SUB VARCHAR2(15), NOTICE VARCHAR2(4000), DATE_TIME DATE, FOREIGN KEY(TNAME, BNAME) REFERENCES BATCHES(TNAME, BNAME))

CREATE TABLE BATCHES(TNAME VARCHAR2(64), BNAME VARCHAR2(64), YEAR NUMBER(4,0), BATCH VARCHAR2(3), BNO NUMBER(3,0), ST_ROLL NUMBER(3), EN_ROLL NUMBER(3), primary key (TNAME, BNAME))

CREATE TABLE UPLOADED_FILES(TNAME VARCHAR2(64), FILENAME VARCHAR2(50), DATE_TIME DATE)

CREATE TABLE SHARED(TNAME VARCHAR2(64), FILENAME VARCHAR2(50), BNAME VARCHAR2(64), DATE_TIME DATE, FOREIGN KEY(TNAME, BNAME) REFERENCES BATCHES(TNAME, BNAME), PRIMARY KEY (TNAME, FILENAME, BNAME))



#MySQL

CREATE TABLE USERS(UNAME VARCHAR(64) PRIMARY KEY NOT NULL, PWD VARCHAR(170) NOT NULL, NAME VARCHAR(64) NOT NULL, SEX CHAR NOT NULL, DOB DATE NOT NULL, EMAIL VARCHAR(320) UNIQUE NOT NULL, MOBILE_NO INT NOT NULL, D CHAR NOT NULL);

CREATE TABLE STUDENTS(UNAME VARCHAR(64) NOT NULL, YEAR INT, BATCH VARCHAR(3), BNO INT, ROLLNO INT, PRIMARY KEY (YEAR, BATCH, BNO, ROLLNO));

CREATE TABLE BATCHES(TNAME VARCHAR(64), BNAME VARCHAR(64), YEAR INT NOT NULL, BATCH VARCHAR(3) NOT NULL, BNO INT, ST_ROLL INT NOT NULL, EN_ROLL INT NOT NULL, PRIMARY KEY (TNAME, BNAME));

CREATE TABLE TEACHERS(TNAME VARCHAR(64) NOT NULL, BNAME VARCHAR(64) NOT NULL, SUB VARCHAR(15), NOTICE VARCHAR(4000) NOT NULL, DATE_TIME DATE NOT NULL, FOREIGN KEY(TNAME, BNAME) REFERENCES BATCHES(TNAME, BNAME) ON UPDATE CASCADE ON DELETE RESTRICT);

CREATE TABLE UPLOADED_FILES(TNAME VARCHAR(64) NOT NULL, FILENAME VARCHAR(50) NOT NULL, DATE_TIME DATE NOT NULL);

CREATE TABLE SHARED(TNAME VARCHAR(64), FILENAME VARCHAR(50), BNAME VARCHAR(64), DATE_TIME DATE, FOREIGN KEY(TNAME, BNAME) REFERENCES BATCHES(TNAME, BNAME) ON UPDATE CASCADE ON DELETE RESTRICT, PRIMARY KEY (TNAME, FILENAME, BNAME));
