CREATE TABLE USERS(UNAME VARCHAR2(64) PRIMARY KEY NOT NULL, PWD VARCHAR2(170), NAME VARCHAR(64), SEX CHAR, DOB DATE, EMAIL VARCHAR2(320) UNIQUE, MOBILE_NO NUMBER(10,0), D CHAR)

CREATE TABLE STUDENTS(UNAME VARCHAR2(64), YEAR NUMBER(4,0), BATCH VARCHAR2(3), BNO NUMBER(3,0), ROLLNO NUMBER(3), constraint PK_D primary key (YEAR, BATCH, BNO, ROLLNO))

CREATE TABLE TEACHERS(TNAME VARCHAR2(64), BNAME VARCHAR2(64), SUB VARCHAR2(15), NOTICE VARCHAR2(4000), DATE_TIME DATE, FOREIGN KEY(TNAME, BNAME) REFERENCES BATCHES(TNAME, BNAME))

CREATE TABLE BATCHES(TNAME VARCHAR2(64), BNAME VARCHAR2(64), YEAR NUMBER(4,0), BATCH VARCHAR2(3), BNO NUMBER(3,0), ST_ROLL NUMBER(3), EN_ROLL NUMBER(3), primary key (TNAME, BNAME))

CREATE TABLE UPLOADED_FILES(TNAME VARCHAR2(64), FILENAME VARCHAR2(50), DATE_TIME DATE)

CREATE TABLE SHARED(TNAME VARCHAR2(64), FILENAME VARCHAR2(50), BNAME VARCHAR2(64), DATE_TIME DATE, FOREIGN KEY(TNAME, BNAME) REFERENCES BATCHES(TNAME, BNAME), PRIMARY KEY (TNAME, FILENAME, BNAME))
